```sql

FROM T_WEBKOZA_APPLY_MASTER WITH(NOLOCK)
WHERE COMPANY_CODE = 'CPS'
AND YEAR(REGISTRATION_DATETIME) = YEAR(DATEADD(month,-1,GETDATE()))
AND MONTH(REGISTRATION_DATETIME) = MONTH(DATEADD(month,-1,GETDATE()));
```
